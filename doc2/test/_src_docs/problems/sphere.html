Sphere function¶

\[\sum\limits_{i=1}^dx_i^2,\quad-10\leq x_i\leq 10,\quad\text{ for }i=1,\ldots,d.\]

Usage¶

import numpy as np
import matplotlib.pyplot as plt

from smt.problems import Sphere

ndim = 2
problem = Sphere(ndim=ndim)

num = 100
x = np.ones((num, ndim))
x[:, 0] = np.linspace(-10, 10., num)
x[:, 1] = 0.
y = problem(x)

yd = np.empty((num, ndim))
for i in range(ndim):
    yd[:, i] = problem(x, kx=i).flatten()

print(y.shape)
print(yd.shape)

plt.plot(x[:, 0], y[:, 0])
plt.xlabel('x')
plt.ylabel('y')
plt.show()

(100, 1)
(100, 2)
